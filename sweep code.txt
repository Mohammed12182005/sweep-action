#include <Servo.h>

Servo servo1, servo2, servo3, servo4;

void setup() {
  servo1.attach(9);
  servo2.attach(10);
  servo3.attach(11);
  servo4.attach(8);

  
  unsigned long start = millis();
  int pos = 0;
  int step = 1;

  while (millis() - start < 2000) {
    servo1.write(pos);
    servo2.write(pos);
    servo3.write(pos);
    servo4.write(pos);

    pos += step;
    if (pos >= 180 || pos <= 0) step = -step;

    delay(5);
  }

  
  servo1.write(90);
  servo2.write(90);
  servo3.write(90);
  servo4.write(90);
}

void loop() {
  
}
